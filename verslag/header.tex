% Index, x, y, r
\def\circle[#1,#2,#3,#4] {
  % The middlepoint
  \coordinate (C#1) at (#2,#3);
  
  % The point
  \draw[fill,color=black] (C#1) circle (1.5pt)
  node[left, yshift=-10pt, color=black] {$C_#1$};
  
  % The radius
  \draw[dotted] (C#1) -- ($(C#1) + (#4,0)$)
  node[left, yshift=-10pt, color=black] {$r_#1$};
  
  
  % The circle
  \draw[thick,color=black] (C#1) circle (#4);
  
}
